from fastapi import FastAPI
from pydantic import BaseModel

app = FastAPI()


class Curso(BaseModel):
    nombre: str
    horas: int
    modalidad: str


cursos = []


@app.post("/cursos")
def crear_curso(curso: Curso):
    cursos.append(curso)
    return {"total": len(cursos)}


@app.get("/cursos")
def listar_cursos(modalidad: str | None = None):
    resultados = [c for c in cursos if modalidad in (None, c.modalidad)]
    return resultados

Líneas

Una instrucción por línea

# Ejemplos de sintaxis básica
def calcular_descuento(precio, porcentaje):
    """Esta función calcula el descuento de un precio"""
    descuento = precio * (porcentaje / 100)
    precio_final = precio - descuento
    return precio_final

# Uso de la función con valores
precio_original = 100
descuento = 20
precio_con_descuento = calcular_descuento(precio_original, descuento)
print(f"Precio final con {descuento}% de descuento: ${precio_con_descuento}")

Estructuras de control

Las estructuras de control determinan el flujo de ejecución de tu programa. Python ofrece una sintaxis clara y concisa para cada tipo de control.

if/elif/else

Bucles con condición

# Ejemplos de estructuras de control

# if/elif/else
def calificar_nota(nota):
    if nota >= 90:
        return "Sobresaliente"
    elif nota >= 70:
        return "Notable"
    elif nota >= 60:
        return "Aprobado"
    else:
        return "Necesita mejorar"

# Bucle for con range y enumerate
print("Tabla de multiplicar del 5:")
for i in range(1, 11):
    print(f"5 x {i} = {5 * i}")

# Bucle while con break y continue
numero = 0
while True:
    numero += 1
    if numero == 3:
        continue  # Salta esta iteración
    if numero > 5:
        break    # Sale del bucle
    print(numero)

Funciones

Las funciones son bloques de código reutilizables que pueden tomar argumentos y devolver resultados.

Argumentos

Parámetros de entrada

Return

Valor de retorno

Args/Kwargs

Argumentos variables

# Ejemplos de funciones

# Función con argumentos por defecto
def saludar(nombre, saludo="¡Hola"):
    return f"{saludo}, {nombre}!"

# Función con argumentos variables
def sumar(*args, **kwargs):
    """
    args: tupla de argumentos posicionales
    kwargs: diccionario de argumentos nombrados
    """
    total = sum(args)  # suma los argumentos posicionales
    if 'extra' in kwargs:
        total += kwargs['extra']
    return total

# Ejemplos de uso
print(saludar("Ana"))  # ¡Hola, Ana!
print(saludar("Juan", "Buenos días"))  # Buenos días, Juan!

# Usando args y kwargs
print(sumar(1, 2, 3))  # 6
print(sumar(1, 2, extra=3))  # 6

Manejo de archivos

Python proporciona una forma sencilla y segura de trabajar con archivos usando el contexto 'with'.

Escritura

Modo 'w' para escribir

Lectura

Modo 'r' para leer

Append

Modo 'a' para añadir

# Ejemplos de manejo de archivos

# Escribir en un archivo
def guardar_notas(archivo, notas):
    with open(archivo, 'w', encoding='utf-8') as f:
        for estudiante, nota in notas.items():
            f.write(f"{estudiante}: {nota}\n")

# Leer de un archivo
def leer_notas(archivo):
    notas = {}
    with open(archivo, 'r', encoding='utf-8') as f:
        for linea in f:
            estudiante, nota = linea.strip().split(': ')
            notas[estudiante] = float(nota)
    return notas

# Ejemplo de uso
notas_estudiantes = {
    'Ana': 9.5,
    'Juan': 8.7,
    'María': 9.2
}

# Guardar notas
guardar_notas('notas.txt', notas_estudiantes)

# Leer notas
notas_leidas = leer_notas('notas.txt')
print("Notas leídas:", notas_leidas)

Python soporta paradigmas de programación funcional que hacen el código más declarativo y conciso.

Pequeñas funciones anónimas definidas con la palabra clave lambda.

# Uso de map y filter
numeros = [1, 2, 3, 4, 5]
cuadrados = list(map(lambda x: x**2, numeros))  # [1, 4, 9, 16, 25]
pares = list(filter(lambda x: x % 2 == 0, numeros))  # [2, 4]

# Decorador simple para medir tiempo de ejecución
import time

def cronometro(func):
    def wrapper(*args, **kwargs):
        inicio = time.time()
        resultado = func(*args, **kwargs)
        fin = time.time()
        print(f"'{func.__name__}' tardó {fin - inicio:.4f} segundos.")
        return resultado
    return wrapper

@cronometro
def proceso_largo():

Programación Orientada a Objetos (POO)

La POO en Python permite modelar problemas complejos mediante clases y objetos, fomentando la reutilización y organización del código.

class Vehiculo:
    def __init__(self, marca, modelo):
        self.marca = marca
        self.modelo = modelo
        self.arrancado = False

    def arrancar(self):
        self.arrancado = True
        print(f"El {self.marca} {self.modelo} ha arrancado.")

    def describir(self):
        return f"Vehículo: {self.marca} {self.modelo}"

class Coche(Vehiculo):  # Herencia de Vehiculo
    def __init__(self, marca, modelo, num_puertas):
        super().__init__(marca, modelo)
        self.num_puertas = num_puertas

    def describir(self):  # Polimorfismo (sobrescritura de método)
        return f"Coche de {self.num_puertas} puertas: {self.marca} {self.modelo}"

# Creación de objetos
mi_coche = Coche("Tesla", "Model 3", 4)
mi_coche.arrancar()
print(mi_coche.describir())

APIs y Desarrollo Web Moderno

Frameworks como FastAPI facilitan construir microservicios que dialogan con los frontends HTML/CSS/JavaScript que has creado.

main.py 
from fastapi import FastAPI
from pydantic import BaseModel

class Modulo(BaseModel):
    nombre: str
    horas: int
    curso: str

app = FastAPI()

MODULOS = [Modulo(nombre="Programación", horas=256, curso="DAM1")]

@app.get("/modulos", response_model=list[Modulo])
def listar_modulos():
    return MODULOS

@app.post("/modulos", response_model=Modulo, status_code=201)
def crear_modulo(modulo: Modulo):
    MODULOS.append(modulo)
    return modulo

Concurrencia y Asincronía

El módulo asyncio facilita gestionar tareas concurrentes como peticiones HTTP o procesamiento en paralelo sin bloquear tu aplicación.

descargas_async.py 
import asyncio
import httpx

URLS = [
    "https://jsonplaceholder.typicode.com/todos/1",
    "https://jsonplaceholder.typicode.com/todos/2",
    "https://jsonplaceholder.typicode.com/todos/3",
]

async def descargar(url: str) -> dict:
    async with httpx.AsyncClient(timeout=10) as client:
        resp = await client.get(url)
        resp.raise_for_status()
        return resp.json()

async def main():
    tareas = [asyncio.create_task(descargar(url)) for url in URLS]
    resultados = await asyncio.gather(*tareas)
    for dato in resultados:
        print(dato["title"])

asyncio.run(main())

bridge.py 
import subprocess
import json

def obtener_datos_java():
    proceso = subprocess.run([
        "java", "-jar", "exportador-xml.jar"
    ], capture_output=True, check=True, text=True)
    return proceso.stdout

def transformar_a_json(xml_str: str) -> str:
    import xmltodict
    data = xmltodict.parse(xml_str)
    return json.dumps(data, indent=2, ensure_ascii=False)

if __name__ == "__main__":
    xml = obtener_datos_java()
    print(transformar_a_json(xml))
