Leer archivos

with open("archivo.txt", "a") as archivo:
    archivo.write("Nueva línea\n")

# Escribir lista de líneas
lineas = ["Línea 1\n", "Línea 2\n", "Línea 3\n"]
with open("archivo.txt", "w") as archivo:
    archivo.writelines(lineas)

POO Avanzada

Python soporta herencia, polimorfismo y conceptos avanzados que facilitan la reutilización y organización del código.

Herencia y Polimorfismo

class Vehiculo:
    def __init__(self, marca, modelo):
        self.marca = marca
        self.modelo = modelo
        self.arrancado = False

    def arrancar(self):
        self.arrancado = True
        print(f"El {self.marca} {self.modelo} ha arrancado.")

    def describir(self):
        return f"Vehículo: {self.marca} {self.modelo}"

class Coche(Vehiculo):  # Herencia
    def __init__(self, marca, modelo, num_puertas):
        super().__init__(marca, modelo)
        self.num_puertas = num_puertas

    def describir(self):  # Polimorfismo
        return f"Coche de {self.num_puertas} puertas: {self.marca} {self.modelo}"

mi_coche = Coche("Tesla", "Model 3", 4)
mi_coche.arrancar()
print(mi_coche.describir())

Herencia Múltiple y Mixins

class Volador:
    def volar(self):
        print("Volando...")

class Nadador:
    def nadar(self):
        print("Nadando...")

class Pato(Volador, Nadador):  # Herencia múltiple
    def __init__(self, nombre):
        self.nombre = nombre

pato = Pato("Donald")
pato.volar()   # De Volador
pato.nadar()   # De Nadador

Decoradores

Los decoradores son funciones que modifican o envuelven a otras funciones para extender su comportamiento.

@property

Convierte métodos en atributos de solo lectura

@staticmethod

Métodos que no necesitan self o cls

@classmethod

Métodos que reciben la clase como primer argumento

class Persona:
    def __init__(self, nombre, edad):
        self._nombre = nombre
        self._edad = edad

    @property
    def nombre(self):
        return self._nombre

    @property
    def edad(self):
        return self._edad

    @staticmethod
    def es_mayor_edad(edad):
        return edad >= 18

    @classmethod
    def desde_string(cls, datos):
        nombre, edad = datos.split(",")
        return cls(nombre, int(edad))

# Uso
p = Persona("Ana", 25)
print(p.nombre)  # Usa @property
print(Persona.es_mayor_edad(25))  # True
p2 = Persona.desde_string("Luis,30")

Decoradores personalizados

import time

def cronometro(func):
    def wrapper(*args, **kwargs):
        inicio = time.time()
        resultado = func(*args, **kwargs)
        fin = time.time()
        print(f"{func.__name__} tardó {fin - inicio:.4f}s")
        return resultado
    return wrapper

@cronometro
def proceso_largo():
    time.sleep(1)
    return "Completado"

resultado = proceso_largo()

Context Managers

Los context managers gestionan recursos de forma segura con la sentencia with.

Métodos especiales

  • __enter__(self) - Se ejecuta al entrar al bloque with
  • __exit__(self, exc_type, exc_val, exc_tb) - Se ejecuta al salir
class GestorArchivo:
    def __init__(self, nombre):
        self.nombre = nombre
        self.archivo = None

    def __enter__(self):
        print(f"Abriendo {self.nombre}")
        self.archivo = open(self.nombre, "w")
        return self.archivo

    def __exit__(self, exc_type, exc_val, exc_tb):
        print(f"Cerrando {self.nombre}")
        if self.archivo:
            self.archivo.close()
        return False

# Uso
with GestorArchivo("datos.txt") as f:
    f.write("Hola desde context manager")
# El archivo se cierra automáticamente

import pytest

    return a + b

    assert sumar(2, 3) == 5

APIs con FastAPI

FastAPI es un framework moderno y rápido para crear APIs web con Python.

from fastapi import FastAPI
from pydantic import BaseModel

app = FastAPI()

class Curso(BaseModel):
    nombre: str
    horas: int
    modalidad: str

cursos = []

@app.get("/")
def inicio():
    return {"mensaje": "API de Cursos DAM"}

@app.post("/cursos")
def crear_curso(curso: Curso):
    cursos.append(curso)
    return {"total": len(cursos), "curso": curso}

@app.get("/cursos")
def listar_cursos(modalidad: str = None):
    if modalidad:
        return [c for c in cursos if c.modalidad == modalidad]
    return cursos

# Ejecutar: uvicorn main:app --reload

Async/Await

El módulo asyncio facilita gestionar tareas concurrentes sin bloquear tu aplicación.

import asyncio
import httpx

async def descargar(url: str):
    async with httpx.AsyncClient() as client:
        resp = await client.get(url)
        return resp.json()

async def main():
    urls = [
        "https://api.example.com/modulo/1",
        "https://api.example.com/modulo/2",
        "https://api.example.com/modulo/3",
    ]
    
    tareas = [asyncio.create_task(descargar(url)) for url in urls]
    resultados = await asyncio.gather(*tareas)
    
    for dato in resultados:
        print(dato)

asyncio.run(main())
